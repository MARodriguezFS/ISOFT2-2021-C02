54  	/**
55  	 * Este metodo se encarga de leer datos de la BD segun la consulta SQL que se le haya proporcionado, y en base al numero de variables que se requieran.
56  	 * @param "SQL_Consulta" el primer parametro del metodo es la consulta sql.
57  	 * @param "vectoradevolver" el segundo parametro del metodo es una lista de objetos.
58  	 * @param "numero_variables" el tercer parametro del metodo es el numero de variables a resolver.
59  	 * @return devuelve la lista de los objetos recogidos en la base de datos.
60  	 * @throws en caso de dar fallo devuelve el fallo obtenido.
61  	 */
62  	public static LinkedList<Object> read(String SQL_Consulta,LinkedList<Object> vectoradevolver, int numero_variables) throws Exception{
63  		
64  		Connection mBD = DriverManager.getConnection(BDConstantes.CONNECTION_STRING,BDConstantes.DBUSER,BDConstantes.DBPASS);	
65  		Statement stmt = mBD.createStatement();
66  		ResultSet res = stmt.executeQuery(SQL_Consulta);
67  		while (res.next()) {
68  			for (int i=1; i<=numero_variables; i++) {
69  				vectoradevolver.add(res.getObject(i));
70  			}
71  		}
72  		stmt.close();
73  		mBD.close();
74  		return vectoradevolver;
75  	}
