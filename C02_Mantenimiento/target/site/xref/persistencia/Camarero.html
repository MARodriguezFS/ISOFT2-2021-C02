36  	 * Este metodo devuelve una lista con todos los camareros del local (con su id y su nombre).
37  	 * @return devuelve una lista de camareros
38  	 * @throws en caso de no hacer bien la operacion se devuelve el fallo
39  	 */
40  	public static LinkedList<Camarero> get_lista_camareros() throws Exception{
41  		AgenteBD agente = AgenteBD.getSingletonInstance();
42  		Camarero camarero = null;
43  		//LinkedList<Object> aux = null;
44  		LinkedList<Camarero> lista_camareros = new LinkedList<Camarero>();
45  		Class.forName("com.mysql.cj.jdbc.Driver"); //.newInstance
46  		LinkedList<Object> vectorADevolver=new LinkedList<Object>();
47  		String SQL_Consulta = "SELECT * FROM Camarero;";
48  		vectorADevolver = agente.read(SQL_Consulta,vectorADevolver, 2);
49  		for(int i =0;i<vectorADevolver.size();i+=2) {
50  		camarero = new Camarero((Integer) vectorADevolver.get(i), (String)vectorADevolver.get(i+1));
51  		lista_camareros.add(camarero);
52  		}
53  		return lista_camareros;
54  		}
