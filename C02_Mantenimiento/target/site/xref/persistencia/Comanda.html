69  	public double cuenta() throws Exception{
70  		double cuenta = 0;
71  		AgenteBD agente = AgenteBD.getSingletonInstance();
72  		Class.forName("com.mysql.cj.jdbc.Driver");
73  		LinkedList<Object> vectorADevolver = new LinkedList<Object>();
74  		String precioBebida = "SELECT precio FROM Bebida WHERE idBebida=(SELECT bebida FROM Comanda WHERE idComanda="+this.idComanda+");";
75  		vectorADevolver = agente.read(precioBebida,vectorADevolver,1);
76  		String precioPrimero = "SELECT precio FROM Comida WHERE idComida=(SELECT primero FROM Comanda WHERE idComanda="+this.idComanda+");";
77  		vectorADevolver = agente.read(precioPrimero,vectorADevolver,1);
78  		String precioSegundo = "SELECT precio FROM Comida WHERE idComida=(SELECT segundo FROM Comanda WHERE idComanda="+this.idComanda+");";
79  		vectorADevolver = agente.read(precioSegundo,vectorADevolver,1);
80  		for(int i = 0; i < vectorADevolver.size(); i+=3) {
81  			cuenta = (Double) vectorADevolver.get(i) + (Double) vectorADevolver.get(i+1) + (Double) vectorADevolver.get(i+2);
82  		}
83  		return cuenta;
84  	}
