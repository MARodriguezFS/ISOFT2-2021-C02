62  		AgenteBD agente = AgenteBD.getSingletonInstance();
63  		Mesa mesa = null;
64  		LinkedList<Mesa> listaMesas = new LinkedList<Mesa>();
65  		Class.forName("com.mysql.cj.jdbc.Driver");
66  		LinkedList<Object> vectorADevolver=new LinkedList<Object>();
67  		String SQL_Consulta = "SELECT * FROM Mesa;";
68  		vectorADevolver = agente.read(SQL_Consulta,vectorADevolver, 3);
69  		for(int i =0;i<vectorADevolver.size();i+=3) {
70  			mesa = new Mesa((Integer) vectorADevolver.get(i), (Integer) vectorADevolver.get(i+1),(String) vectorADevolver.get(i+2));
71  			listaMesas.add(mesa);
72  		}
73  		return listaMesas;
74  	}
